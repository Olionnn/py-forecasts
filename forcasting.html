# Forcasting Data Gempa Bumi Menggunakan Prophet
model = Prophet(interval_width=0.95, daily_seasonality=True)
model.fit(df)
future = model.make_future_dataframe(periods=100,freq='M') # membuat data prediksi 100 Bukan kedepan
forecast = model.predict(future)
forecast[['ds', 'yhat', 'yhat_lower', 'yhat_upper']].tail()
